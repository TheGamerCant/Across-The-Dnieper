POL_get_sejm_seats={
	clear_array = sejm_seat_frame_array
	clear_array = sejm_seat_tooltip_array
	resize_array = {
		array = sejm_seat_frame_array
		value = 1
		size = POL.POL_sejm_seat_count
	}
	resize_array = {
		array = sejm_seat_tooltip_array
		value = 0
		size = POL.POL_sejm_seat_count
	}
	
	set_variable = { current_faction_id = 0 }
	set_variable = { current_faction_seat_count = POL.party_seat_count^0 }
	for_loop_effect = {
		start = 0
		end = POL.POL_sejm_seat_count
		value = i
		
		while_loop_effect = {
			limit = { check_variable = { var = i value = current_faction_seat_count compare = greater_than_or_equals } }
			#log = "Party: [?current_faction_id] has seat count [?i]"
			add_to_variable = { current_faction_id = 1 }
			add_to_variable = { current_faction_seat_count = POL.party_seat_count^current_faction_id }
		}
		
		set_variable = { sejm_seat_tooltip_array^i = current_faction_id }
		set_variable = { sejm_seat_frame_array^i = current_faction_id }
		add_to_variable  = { sejm_seat_frame_array^i = 1 }
	}
	clear_variable = current_faction_id
	clear_variable = current_faction_seat_count
}
