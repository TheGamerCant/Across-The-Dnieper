#Creation of the United Left
country_event = {
	id = poland.2
	title = poland.2.t
	desc = poland.2.desc
	picture = GFX_report_event_POL_united_left
	is_triggered_only = yes

	option = {
		name = poland.2.a
		ai_chance = { base = 1 }

		add_popularity = { ideology = social_democracy popularity = 0.02 }
		add_popularity = { ideology = socialist popularity = 0.005 }
		clr_country_flag = POL_leftist_party_is_democratic_left_alliance
		set_country_flag = POL_leftist_party_is_united_left
		
		hidden_effect={
			every_state={
				limit = { POL_is_election_state = yes }
				multiply_variable = { THIS.OTHER_popularity = 0.5 }
				multiply_variable = { THIS.LEFT_popularity = 1.08 }
				POL_make_popularities_equal_100 = yes
			}
			
			set_party_name = { 
				ideology = social_democracy 
				long_name = POL_ZL_no_colour 
				name =  POL_ZL_no_colour_shortened 
			}
			
			if={
				limit = { bronislaw_komorowski_is_leader = yes }
				create_country_leader = {
					name = "Bronisław Komorowski"
					picture = gfx/leaders/POL/Portrait_Bronisław_Komorowski.png
					ideology = poland_united_left
					
					traits = { POL_former_minister_of_defence POL_liberal_reformer }
				}
			}
			else_if={
				limit = { andrzej_duda_is_leader = yes }
				create_country_leader = {
					name = ""
					picture = gfx/leaders/POL/Portrait_Andrzej_Duda.png
					ideology = poland_united_left
					
					traits = {  }
				}
			}
			else_if={
				limit = { jaroslaw_kaczynski_is_leader = yes }
				create_country_leader = {
					name = ""
					picture = gfx/leaders/POL/Portrait_Jarosław_Kaczyński.png
					ideology = poland_united_left
					
					traits = { POL_conservative_powerhouse }
				}
			}
		}
		
		custom_effect_tooltip = make_united_left_alliance_tt
	}
}
